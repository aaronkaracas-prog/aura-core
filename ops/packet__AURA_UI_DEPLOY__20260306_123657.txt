<script>
async function sendCmd(){
let cmd=document.getElementById('cmd').value;
let res=await fetch('/chat',{method:'POST',body:cmd});
let txt=await res.text();
document.getElementById('console').innerHTML+='<pre>'+txt+'</pre>';
}
function startMic(){
const r=new webkitSpeechRecognition();
r.onresult=e=>{document.getElementById('cmd').value=e.results[0][0].transcript}
r.start();
}
</script>
